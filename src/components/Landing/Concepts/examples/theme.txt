import { render } from 'react-dom'
import {
  CandourProvider,
  Text,
} from 'candour'
import colors from 'candour-colors'
import fluidSteps from 'candour-fluid-steps'

const theme = {
  text: {
    base: {
      backgroundColor: 'primary',
      fontWeight: 700,
      padding: 1,
    }
  }
}

render(
  <CandourProvider
    converters={[colors, fluidSteps]}
    colors={{ primary: '#66fcd9' }}
    theme={theme}
  >
    <Text>
      It's magic
    </Text>
  </CandourProvider>
)
